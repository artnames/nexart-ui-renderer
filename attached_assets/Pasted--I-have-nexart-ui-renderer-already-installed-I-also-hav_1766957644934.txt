// GENRE: Generative Abstraction

function setup() {
  noFill();
}

function draw() {
  const totalFrames = 120;
  const t = (frameCount % totalFrames) / totalFrames;

  background(246, 245, 242); // off-white

  push();
  translate(width * 0.5, height * 0.5);
  rotate(radians(148));

  const span = max(width, height) * 6.2;
  const layers = 7;
  const points = 460;
  const spacing = min(width, height) * 0.025;

  stroke(55, 55, 55); // dark grey
  strokeWeight(2);

  for (let l = 0; l < layers; l++) {
    const yBase = (l - (layers - 1) / 2) * spacing;
    const freq = 1.2 + l * 0.45;
    const phase = l * 0.5;

    beginShape();
    for (let i = 0; i < points; i++) {
      const x = map(i, 0, points - 1, -span / 2, span / 2);
      const wave =
        sin((i / points) * TWO_PI * freq + t * TWO_PI + phase) *
        cos(t * TWO_PI);
      const amp = min(width, height) * 0.04 * (0.6 + l * 0.08);
      vertex(x, yBase + wave * amp);
    }
    endShape();
  }

  pop();
}
